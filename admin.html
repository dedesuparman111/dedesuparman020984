💡 Data tersimpan di tabel link_hub Supabase

Setup Database SQL

Buat tabel link_hub di Supabase → SQL Editor:

-- Tabel Link Hub CREATE TABLE IF NOT EXISTS link_hub ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, name text NOT NULL, url text NOT NULL, icon text DEFAULT '🔗', sort_order integer DEFAULT 0, is_active boolean DEFAULT true, created_at timestamptz DEFAULT now() ); CREATE INDEX IF NOT EXISTS link_hub_sort_idx ON link_hub (sort_order ASC); ALTER TABLE link_hub ENABLE ROW LEVEL SECURITY; CREATE POLICY "Public read link_hub" ON link_hub FOR SELECT USING (true); CREATE POLICY "Auth write link_hub" ON link_hub FOR ALL USING (auth.role() = 'authenticated') WITH CHECK (auth.role() = 'authenticated');
